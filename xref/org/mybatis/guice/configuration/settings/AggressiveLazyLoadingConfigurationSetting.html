13  package org.mybatis.guice.configuration.settings;
14  
15  import org.apache.ibatis.session.Configuration;
16  
17  public class AggressiveLazyLoadingConfigurationSetting implements ConfigurationSetting {
18  
19    private final boolean aggressiveLazyLoading;
20  
21    public AggressiveLazyLoadingConfigurationSetting(final boolean aggressiveLazyLoading) {
22      this.aggressiveLazyLoading = aggressiveLazyLoading;
23    }
24  
25    @Override
26    public void applyConfigurationSetting(Configuration configuration) {
27      configuration.setAggressiveLazyLoading(aggressiveLazyLoading);
28    }
29  }
