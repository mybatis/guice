16  package org.mybatis.guice.configuration;
17  
18  import com.google.inject.ProvisionException;
19  
20  import edu.umd.cs.findbugs.annotations.Nullable;
21  
22  import jakarta.inject.Named;
23  import jakarta.inject.Provider;
24  import jakarta.inject.Singleton;
25  
26  import java.util.HashSet;
27  import java.util.Set;
28  
29  import javax.sql.DataSource;
30  
31  import org.apache.ibatis.executor.ErrorContext;
32  import org.apache.ibatis.mapping.DatabaseIdProvider;
33  import org.apache.ibatis.mapping.Environment;
34  import org.apache.ibatis.session.AutoMappingBehavior;
35  import org.apache.ibatis.session.Configuration;
36  import org.apache.ibatis.session.ExecutorType;
37  import org.mybatis.guice.configuration.settings.ConfigurationSetting;
38  import org.mybatis.guice.configuration.settings.MapperConfigurationSetting;
39  
40  /**
41   * Provides the myBatis Configuration.
42   */
43  @Singleton
44  public class ConfigurationProvider implements Provider<Configuration>, ConfigurationSettingListener {
45  
46    /**
47     * The myBatis Configuration reference.
48     */
49    private final Environment environment;
50  
51    @com.google.inject.Inject(optional = true)
52    @Named("mybatis.configuration.lazyLoadingEnabled")
53    private boolean lazyLoadingEnabled;
54  
55    @com.google.inject.Inject(optional = true)
56    @Named("mybatis.configuration.aggressiveLazyLoading")
57    private boolean aggressiveLazyLoading = true;
58  
59    @Deprecated
60    @com.google.inject.Inject(optional = true)
61    @Named("mybatis.configuration.multipleResultSetsEnabled")
62    private boolean multipleResultSetsEnabled = true;
63  
64    @com.google.inject.Inject(optional = true)
65    @Named("mybatis.configuration.useGeneratedKeys")
66    private boolean useGeneratedKeys;
67  
68    @com.google.inject.Inject(optional = true)
69    @Named("mybatis.configuration.useColumnLabel")
70    private boolean useColumnLabel = true;
71  
72    @com.google.inject.Inject(optional = true)
73    @Named("mybatis.configuration.cacheEnabled")
74    private boolean cacheEnabled = true;
75  
76    @com.google.inject.Inject(optional = true)
77    @Named("mybatis.configuration.defaultExecutorType")
78    private ExecutorType defaultExecutorType = ExecutorType.SIMPLE;
79  
80    @com.google.inject.Inject(optional = true)
81    @Named("mybatis.configuration.autoMappingBehavior")
82    private AutoMappingBehavior autoMappingBehavior = AutoMappingBehavior.PARTIAL;
83  
84    @com.google.inject.Inject(optional = true)
85    @Named("mybatis.configuration.callSettersOnNulls")
86    private boolean callSettersOnNulls;
87  
88    @com.google.inject.Inject(optional = true)
89    @Named("mybatis.configuration.defaultStatementTimeout")
90    @Nullable
91    private Integer defaultStatementTimeout;
92  
93    @com.google.inject.Inject(optional = true)
94    @Named("mybatis.configuration.mapUnderscoreToCamelCase")
95    private boolean mapUnderscoreToCamelCase;
96  
97    @com.google.inject.Inject(optional = true)
98    @Named("mybatis.configuration.failFast")
99    private boolean failFast;
100 
101   @com.google.inject.Inject(optional = true)
102   private DatabaseIdProvider databaseIdProvider;
103 
104   @com.google.inject.Inject
105   private DataSource dataSource;
106 
107   private Set<ConfigurationSetting> configurationSettings = new HashSet<>();
108   private Set<MapperConfigurationSetting> mapperConfigurationSettings = new HashSet<>();
109 
110   /**
111    * Instantiates a new configuration provider.
112    *
113    * @param environment
114    *          the environment
115    *
116    * @since 1.0.1
117    */
118   @com.google.inject.Inject
119   public ConfigurationProvider(final Environment environment) {
120     this.environment = environment;
121   }
122 
123   @Deprecated
124   public void setEnvironment(Environment environment) {
125     // do nothing
126   }
127 
128   /**
129    * Flag to check all statements are completed.
130    *
131    * @param failFast
132    *          flag to check all statements are completed
133    *
134    * @since 1.0.1
135    */
136   public void setFailFast(boolean failFast) {
137     this.failFast = failFast;
138   }
139 
140   @Override
141   public void addConfigurationSetting(ConfigurationSetting configurationSetting) {
142     this.configurationSettings.add(configurationSetting);
143   }
144 
145   @Override
146   public void addMapperConfigurationSetting(MapperConfigurationSetting mapperConfigurationSetting) {
147     this.mapperConfigurationSettings.add(mapperConfigurationSetting);
148   }
149 
150   /**
151    * New configuration.
152    *
153    * @param environment
154    *          the environment
155    *
156    * @return new configuration
157    */
158   protected Configuration newConfiguration(Environment environment) {
159     return new Configuration(environment);
160   }
161 
162   @Override
163   public Configuration get() {
164     final Configuration configuration = newConfiguration(environment);
165     configuration.setLazyLoadingEnabled(lazyLoadingEnabled);
166     configuration.setAggressiveLazyLoading(aggressiveLazyLoading);
167     configuration.setUseGeneratedKeys(useGeneratedKeys);
168     configuration.setUseColumnLabel(useColumnLabel);
169     configuration.setCacheEnabled(cacheEnabled);
170     configuration.setDefaultExecutorType(defaultExecutorType);
171     configuration.setAutoMappingBehavior(autoMappingBehavior);
172     configuration.setCallSettersOnNulls(callSettersOnNulls);
173     configuration.setDefaultStatementTimeout(defaultStatementTimeout);
174     configuration.setMapUnderscoreToCamelCase(mapUnderscoreToCamelCase);
175 
176     for (ConfigurationSetting setting : configurationSettings) {
177       setting.applyConfigurationSetting(configuration);
178     }
179 
180     try {
181       if (databaseIdProvider != null) {
182         configuration.setDatabaseId(databaseIdProvider.getDatabaseId(dataSource));
183       }
184 
185       for (MapperConfigurationSetting setting : mapperConfigurationSettings) {
186         setting.applyConfigurationSetting(configuration);
187       }
188 
189       if (failFast) {
190         configuration.getMappedStatementNames();
191       }
192     } catch (Throwable cause) {
193       throw new ProvisionException("An error occurred while building the org.apache.ibatis.session.Configuration",
194           cause);
195     } finally {
196       ErrorContext.instance().reset();
197     }
198 
199     return configuration;
200   }
201 }
