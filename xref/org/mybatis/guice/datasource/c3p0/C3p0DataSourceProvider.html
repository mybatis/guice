33    /**
34     * The ComboPooledDataSource reference.
35     */
36    private final ComboPooledDataSource dataSource = new ComboPooledDataSource();
37    private String username;
38    private String password;

40    /**
41     * Creates a new ComboPooledDataSource using the needed parameter.
42     *
43     * @param driver
44     *          The JDBC driver class.
45     * @param url
46     *          the database URL of the form <code>jdbc:subprotocol:subname</code>.
47     */
48    @Inject
49    public C3p0DataSourceProvider(@Named("JDBC.driver") final String driver, @Named("JDBC.url") final String url) {
50      try {
51        dataSource.setDriverClass(driver);
52      } catch (PropertyVetoException e) {
53        throw new RuntimeException(
54            "Impossible to initialize C3P0 Data Source with driver class '" + driver + "', see nested exceptions", e);
55      }
56      dataSource.setJdbcUrl(url);
57    }
