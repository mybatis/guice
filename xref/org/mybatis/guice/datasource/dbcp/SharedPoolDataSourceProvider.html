13  package org.mybatis.guice.datasource.dbcp;
14  
15  import com.google.inject.Provider;
16  
17  import org.apache.commons.dbcp.datasources.SharedPoolDataSource;
18  
19  import javax.inject.Named;
20  import javax.sql.ConnectionPoolDataSource;
21  import javax.sql.DataSource;
22  
23  /**
24   * Provides the Apache commons-dbcp {@code SharedPoolDataSource}.
25   */
26  public final class SharedPoolDataSourceProvider implements Provider<DataSource> {
27  
28    private final SharedPoolDataSource dataSource = new SharedPoolDataSource();
29  
30    @com.google.inject.Inject(optional = true)
31    public void setConnectionPoolDataSource(ConnectionPoolDataSource cpds) {
32      dataSource.setConnectionPoolDataSource(cpds);
33    }
34  
35    @com.google.inject.Inject(optional = true)
36    public void setDataSourceName(@Named("DBCP.name") String name) {
37      dataSource.setDataSourceName(name);
38    }
39  
40    @com.google.inject.Inject(optional = true)
41    public void setDefaultAutoCommit(@Named("JDBC.autoCommit") boolean autoCommit) {
42      dataSource.setDefaultAutoCommit(autoCommit);
43    }
44  
45    @com.google.inject.Inject(optional = true)
46    public void setDefaultReadOnly(@Named("DBCP.defaultReadOnly") boolean defaultReadOnly) {
47      dataSource.setDefaultReadOnly(defaultReadOnly);
48    }
49  
50    @com.google.inject.Inject(optional = true)
51    public void setDefaultTransactionIsolation(
52        @Named("DBCP.defaultTransactionIsolation") int defaultTransactionIsolation) {
53      dataSource.setDefaultTransactionIsolation(defaultTransactionIsolation);
54    }
55  
56    @com.google.inject.Inject(optional = true)
57    public void setDescription(@Named("DBCP.description") String description) {
58      dataSource.setDescription(description);
59    }
60  
61    @com.google.inject.Inject(optional = true)
62    public void setJndiEnvironment(@Named("DBCP.jndi.key") String key, @Named("DBCP.jndi.value") String value) {
63      dataSource.setJndiEnvironment(key, value);
64    }
65  
66    @com.google.inject.Inject(optional = true)
67    public void setLoginTimeout(@Named("JDBC.loginTimeout") int loginTimeout) {
68      dataSource.setLoginTimeout(loginTimeout);
69    }
70  
71    @com.google.inject.Inject(optional = true)
72    public void setMinEvictableIdleTimeMillis(@Named("DBCP.minEvictableIdleTimeMillis") int minEvictableIdleTimeMillis) {
73      dataSource.setMinEvictableIdleTimeMillis(minEvictableIdleTimeMillis);
74    }
75  
76    @com.google.inject.Inject(optional = true)
77    public void setNumTestsPerEvictionRun(@Named("DBCP.numTestsPerEvictionRun") int numTestsPerEvictionRun) {
78      dataSource.setNumTestsPerEvictionRun(numTestsPerEvictionRun);
79    }
80  
81    @com.google.inject.Inject(optional = true)
82    public void setRollbackAfterValidation(@Named("DBCP.rollbackAfterValidation") boolean rollbackAfterValidation) {
83      dataSource.setRollbackAfterValidation(rollbackAfterValidation);
84    }
85  
86    @com.google.inject.Inject(optional = true)
87    public void setTestOnBorrow(@Named("DBCP.testOnBorrow") boolean testOnBorrow) {
88      dataSource.setTestOnBorrow(testOnBorrow);
89    }
90  
91    @com.google.inject.Inject(optional = true)
92    public void setTestOnReturn(@Named("DBCP.testOnReturn") boolean testOnReturn) {
93      dataSource.setTestOnReturn(testOnReturn);
94    }
95  
96    @com.google.inject.Inject(optional = true)
97    public void setTestWhileIdle(@Named("DBCP.testWhileIdle") boolean testWhileIdle) {
98      dataSource.setTestWhileIdle(testWhileIdle);
99    }
100 
101   @com.google.inject.Inject(optional = true)
102   public void setTimeBetweenEvictionRunsMillis(
103       @Named("DBCP.timeBetweenEvictionRunsMillis") int timeBetweenEvictionRunsMillis) {
104     dataSource.setTimeBetweenEvictionRunsMillis(timeBetweenEvictionRunsMillis);
105   }
106 
107   @com.google.inject.Inject(optional = true)
108   public void setValidationQuery(@Named("DBCP.validationQuery") String validationQuery) {
109     dataSource.setValidationQuery(validationQuery);
110   }
111 
112   /**
113    * Sets the max active.
114    *
115    * @param maxActive
116    *          the new max active
117    */
118   @com.google.inject.Inject(optional = true)
119   public void setMaxActive(@Named("DBCP.maxActive") final int maxActive) {
120     dataSource.setMaxActive(maxActive);
121   }
122 
123   /**
124    * Sets the max idle.
125    *
126    * @param maxIdle
127    *          the new max idle
128    */
129   @com.google.inject.Inject(optional = true)
130   public void setMaxIdle(@Named("DBCP.maxIdle") final int maxIdle) {
131     dataSource.setMaxIdle(maxIdle);
132   }
133 
134   /**
135    * Sets the max wait.
136    *
137    * @param maxWait
138    *          the new max wait
139    */
140   @com.google.inject.Inject(optional = true)
141   public void setMaxWait(@Named("DBCP.maxWait") final int maxWait) {
142     dataSource.setMaxWait(maxWait);
143   }
144 
145   @Override
146   public DataSource get() {
147     return dataSource;
148   }
149 
150 }
