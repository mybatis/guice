13  package org.mybatis.guice.transactional;
14  
15  import org.apache.ibatis.session.TransactionIsolationLevel;
16  
17  /**
18   * Enum of isolation levels.  This enum exists because Java annotations
19   * do not support null default values - so we need to add the DEFAULT
20   * level which means - do not specify an isolation level.
21   *
22   * @author Jeff Butler
23   * @since 3.1
24   */
25  public enum Isolation {
26  
27    DEFAULT(null), NONE(TransactionIsolationLevel.NONE), READ_COMMITTED(
28        TransactionIsolationLevel.READ_COMMITTED), READ_UNCOMMITTED(
29            TransactionIsolationLevel.READ_UNCOMMITTED), REPEATABLE_READ(
30                TransactionIsolationLevel.REPEATABLE_READ), SERIALIZABLE(TransactionIsolationLevel.SERIALIZABLE);
31  
32    private final TransactionIsolationLevel transactionIsolationLevel;
33  
34    private Isolation(TransactionIsolationLevel transactionIsolationLevel) {
35      this.transactionIsolationLevel = transactionIsolationLevel;
36    }
37  
38    public TransactionIsolationLevel getTransactionIsolationLevel() {
39      return transactionIsolationLevel;
40    }
41  
42  }
