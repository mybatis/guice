37    @Test
38    public void get() throws Throwable {
39      final String driver = "org.mybatis.guice.TestDriver";
40      final String url = "jdbc:h2:mem:testdb";
41      final String username = "test_user";
42      final String password = "test_password";
43      final boolean autoCommit = true;
44      final Properties driverProperties = new Properties();
45      driverProperties.put("my_property", "true");
46      final boolean accessToUnderlyingConnectionAllowed = true;
47      final String defaultCatalog = "test_catalog";
48      final boolean defaultReadOnly = true;
49      final int defaultTransactionIsolation = 20;
50      final int initialSize = 30;
51      final int maxActive = 40;

60      final boolean testWhileIdle = true;
61      final int timeBetweenEvictionRunsMillis = 100;
62      final String validationQuery = "SELECT 1";
63      Injector injector = Guice.createInjector(new AbstractModule() {
64        @Override
65        protected void configure() {
66          bind(ClassLoader.class).annotatedWith(Names.named("JDBC.driverClassLoader")).toInstance(driverClassLoader);
67          bindConstant().annotatedWith(Names.named("JDBC.driver")).to(driver);
68          bindConstant().annotatedWith(Names.named("JDBC.url")).to(url);
69          bindConstant().annotatedWith(Names.named("JDBC.username")).to(username);
70          bindConstant().annotatedWith(Names.named("JDBC.password")).to(password);
71          bindConstant().annotatedWith(Names.named("JDBC.autoCommit")).to(autoCommit);
72          bind(Properties.class).annotatedWith(Names.named("JDBC.driverProperties")).toInstance(driverProperties);
73          bindConstant().annotatedWith(Names.named("DBCP.accessToUnderlyingConnectionAllowed"))
74              .to(accessToUnderlyingConnectionAllowed);
75          bindConstant().annotatedWith(Names.named("DBCP.defaultCatalog")).to(defaultCatalog);
76          bindConstant().annotatedWith(Names.named("DBCP.defaultReadOnly")).to(defaultReadOnly);
77          bindConstant().annotatedWith(Names.named("DBCP.defaultTransactionIsolation")).to(defaultTransactionIsolation);
78          bindConstant().annotatedWith(Names.named("DBCP.initialSize")).to(initialSize);
79          bindConstant().annotatedWith(Names.named("DBCP.maxActive")).to(maxActive);
80          bindConstant().annotatedWith(Names.named("DBCP.maxIdle")).to(maxIdle);
81          bindConstant().annotatedWith(Names.named("DBCP.maxOpenPreparedStatements")).to(maxOpenPreparedStatements);
82          bindConstant().annotatedWith(Names.named("DBCP.maxWait")).to(maxWait);
83          bindConstant().annotatedWith(Names.named("DBCP.minIdle")).to(minIdle);
84          bindConstant().annotatedWith(Names.named("DBCP.numTestsPerEvictionRun")).to(numTestsPerEvictionRun);
85          bindConstant().annotatedWith(Names.named("DBCP.poolPreparedStatements")).to(poolPreparedStatements);
86          bindConstant().annotatedWith(Names.named("DBCP.testOnBorrow")).to(testOnBorrow);
87          bindConstant().annotatedWith(Names.named("DBCP.testOnReturn")).to(testOnReturn);
88          bindConstant().annotatedWith(Names.named("DBCP.testWhileIdle")).to(testWhileIdle);
89          bindConstant().annotatedWith(Names.named("DBCP.timeBetweenEvictionRunsMillis"))
90              .to(timeBetweenEvictionRunsMillis);
91          bindConstant().annotatedWith(Names.named("DBCP.validationQuery")).to(validationQuery);
92        }
93      });
94      BasicDataSourceProvider provider = injector.getInstance(BasicDataSourceProvider.class);
95  
96      BasicDataSource dataSource = (BasicDataSource) provider.get();
97  
98      assertEquals(driverClassLoader, dataSource.getDriverClassLoader());
99      assertEquals(driver, dataSource.getDriverClassName());
100     assertEquals(url, dataSource.getUrl());
101     assertEquals(username, dataSource.getUsername());
102     assertEquals(password, dataSource.getPassword());
103     assertEquals(autoCommit, dataSource.getDefaultAutoCommit());
104     // Cannot test driver properties.
105     assertEquals(accessToUnderlyingConnectionAllowed, dataSource.isAccessToUnderlyingConnectionAllowed());
106     assertEquals(defaultCatalog, dataSource.getDefaultCatalog());
107     assertEquals(defaultReadOnly, dataSource.getDefaultReadOnly());
108     assertEquals(defaultTransactionIsolation, dataSource.getDefaultTransactionIsolation());
109     assertEquals(initialSize, dataSource.getInitialSize());
110     assertEquals(maxActive, dataSource.getMaxActive());
111     assertEquals(maxIdle, dataSource.getMaxIdle());
112     assertEquals(maxOpenPreparedStatements, dataSource.getMaxOpenPreparedStatements());
113     assertEquals(maxWait, dataSource.getMaxWait());
114     assertEquals(minIdle, dataSource.getMinIdle());
115     assertEquals(numTestsPerEvictionRun, dataSource.getNumTestsPerEvictionRun());
116     assertEquals(poolPreparedStatements, dataSource.isPoolPreparedStatements());
117     assertEquals(testOnBorrow, dataSource.getTestOnBorrow());
118     assertEquals(testOnReturn, dataSource.getTestOnReturn());
119     assertEquals(testWhileIdle, dataSource.getTestWhileIdle());
120     assertEquals(timeBetweenEvictionRunsMillis, dataSource.getTimeBetweenEvictionRunsMillis());
121     assertEquals(validationQuery, dataSource.getValidationQuery());
