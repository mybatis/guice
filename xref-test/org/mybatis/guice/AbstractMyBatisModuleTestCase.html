16  package org.mybatis.guice;
17  
18  import static org.junit.jupiter.api.Assertions.assertNotNull;
19  
20  import jakarta.inject.Inject;
21  
22  import java.util.List;
23  
24  import org.junit.jupiter.api.Assertions;
25  import org.junit.jupiter.api.Test;
26  
27  public abstract class AbstractMyBatisModuleTestCase {
28  
29    @Inject
30    private Contact contact;
31  
32    @Inject
33    private ContactMapperClient contactMapperClient;
34  
35    @Inject
36    private Counter counter;
37  
38    public void setContact(Contact contact) {
39      this.contact = contact;
40    }
41  
42    public void setContactMapperClient(ContactMapperClient contactMapperClient) {
43      this.contactMapperClient = contactMapperClient;
44    }
45  
46    @Test
47    void verifyNotNullMapper() {
48      assertNotNull(this.contactMapperClient);
49    }
50  
51    @Test
52    void insertContact() throws Exception {
53      this.contactMapperClient.insert(this.contact);
54    }
55  
56    @Test
57    void selectAllContacts() throws Exception {
58      List<Contact> contacts = this.contactMapperClient.getAll();
59      assert contacts.size() > 0 : "Expected not empty contact table";
60    }
61  
62    @Test
63    void reSelectAllContacts() throws Exception {
64      List<Contact> contacts = this.contactMapperClient.getAll();
65      assert contacts.size() > 0 : "Expected not empty contact table";
66    }
67  
68    @Test
69    void selectContact() throws Exception {
70      Contact contact = this.contactMapperClient.selectById(this.contact.getId());
71      assert contact != null : "impossible to retrieve Contact with id '" + this.contact.getId() + "'";
72      assert this.contact.equals(contact) : "Expected " + this.contact + " but found " + contact;
73    }
74  
75    @Test
76    void selectContactWithTypeHandler() throws Exception {
77      Contact contact = this.contactMapperClient.selectByIdWithTypeHandler(this.contact.getId());
78      assert contact != null : "impossible to retrieve Contact with id '" + this.contact.getId() + "'";
79      assert this.contact.equals(contact) : "Expected " + this.contact + " but found " + contact;
80    }
81  
82    @Test
83    void catchSQLException() throws Exception {
84      Assertions.assertThrows(CustomException.class, () -> {
85        this.contactMapperClient.brokenInsert(this.contact);
86      });
87    }
88  
89    @Test
90    void testCountInterceptor() throws Exception {
91      counter.reset();
92      assert 0 == counter.getCount() : "Expected 0 update in counter, but was " + counter.getCount();
93      this.contactMapperClient.update(contact);
94      assert 1 == counter.getCount() : "Expected 1 update in Executor, but was " + counter.getCount();
95    }
96  
97  }
