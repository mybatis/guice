118   @Test
119   void get_Optionals() throws Throwable {
120     String databaseId = "test_database_id";
121     final Integer defaultFetchSize = 200;
122     final Integer defaultStatementTimeout = 2000;
123     when(databaseIdProvider.getDatabaseId(dataSource)).thenReturn(databaseId);
124     final Key<TypeHandler<Alias>> aliasTypeHandlerKey = Key.get(new TypeLiteral<TypeHandler<Alias>>() {
125     });
126     injector = Guice.createInjector(new AbstractModule() {
127       @Override
128       protected void configure() {
129         bind(Environment.class).toInstance(environment);
130         bind(DataSource.class).toInstance(dataSource);
131         bindConstant().annotatedWith(Names.named("mybatis.configuration.lazyLoadingEnabled")).to(true);
132         bindConstant().annotatedWith(Names.named("mybatis.configuration.aggressiveLazyLoading")).to(false);
133         bindConstant().annotatedWith(Names.named("mybatis.configuration.multipleResultSetsEnabled")).to(false);
134         bindConstant().annotatedWith(Names.named("mybatis.configuration.useGeneratedKeys")).to(true);
135         bindConstant().annotatedWith(Names.named("mybatis.configuration.useColumnLabel")).to(false);
136         bindConstant().annotatedWith(Names.named("mybatis.configuration.cacheEnabled")).to(false);
137         bindConstant().annotatedWith(Names.named("mybatis.configuration.defaultExecutorType")).to(ExecutorType.REUSE);
138         bindConstant().annotatedWith(Names.named("mybatis.configuration.autoMappingBehavior"))
139             .to(AutoMappingBehavior.FULL);
140         bindConstant().annotatedWith(Names.named("mybatis.configuration.callSettersOnNulls")).to(true);
141         bindConstant().annotatedWith(Names.named("mybatis.configuration.defaultStatementTimeout"))
142             .to(defaultStatementTimeout);
143         bindConstant().annotatedWith(Names.named("mybatis.configuration.mapUnderscoreToCamelCase")).to(true);
144         bind(DatabaseIdProvider.class).toInstance(databaseIdProvider);
145         bind(Configuration.class).toProvider(configurationProvider);
146         bind(aliasTypeHandlerKey).toInstance(aliasTypeHandler);
147         bind(Interceptor.class).toInstance(interceptor);
148       }
149     });
150     configurationProvider.addConfigurationSetting(new AliasConfigurationSetting("alias", Alias.class));
151     JavaTypeAndHandlerConfigurationSettingProvider aliasTypeHandlerSetting = JavaTypeAndHandlerConfigurationSettingProvider
152         .create(Alias.class, aliasTypeHandlerKey);
153     injector.injectMembers(aliasTypeHandlerSetting);
154     configurationProvider.addConfigurationSetting(aliasTypeHandlerSetting.get());
155     TypeHandlerConfigurationSettingProvider humanTypeHandlerSetting = new TypeHandlerConfigurationSettingProvider(
156         Key.get(HumanTypeHandler.class));
157     injector.injectMembers(humanTypeHandlerSetting);
158     configurationProvider.addConfigurationSetting(humanTypeHandlerSetting.get());
159     configurationProvider.addMapperConfigurationSetting(new MapperConfigurationSetting(TestMapper.class));
160     InterceptorConfigurationSettingProvider interceptorSetting = new InterceptorConfigurationSettingProvider(
161         Interceptor.class);
162     injector.injectMembers(interceptorSetting);
163     configurationProvider.addConfigurationSetting(interceptorSetting.get());
164     configurationProvider.addConfigurationSetting((new ConfigurationSetting() {
165       @Override
166       public void applyConfigurationSetting(Configuration configuration) {
167         configuration.setDefaultFetchSize(defaultFetchSize);
168       }
169     }));
170 
171     Configuration configuration = injector.getInstance(Configuration.class);
172 
173     configuration.getMappedStatementNames(); // Test that configuration is valid.
174     assertEquals(environment, configuration.getEnvironment());
175     assertEquals(Alias.class, configuration.getTypeAliasRegistry().getTypeAliases().get("alias"));
176     assertTrue(configuration.getTypeHandlerRegistry().hasTypeHandler(Alias.class));
177     assertTrue(configuration.getTypeHandlerRegistry().hasTypeHandler(Human.class));
178     assertEquals(1, configuration.getMapperRegistry().getMappers().size());
179     assertTrue(configuration.getMapperRegistry().getMappers().contains(TestMapper.class));
180     assertEquals(1, configuration.getInterceptors().size());
181     assertTrue(configuration.getInterceptors().contains(interceptor));
182     verify(databaseIdProvider).getDatabaseId(dataSource);
183     assertEquals(databaseId, configuration.getDatabaseId());
184     assertTrue(configuration.isLazyLoadingEnabled());
185     assertFalse(configuration.isAggressiveLazyLoading());
186     assertFalse(configuration.isMultipleResultSetsEnabled());
187     assertTrue(configuration.isUseGeneratedKeys());
188     assertFalse(configuration.isUseColumnLabel());
189     assertFalse(configuration.isCacheEnabled());
190     assertEquals(ExecutorType.REUSE, configuration.getDefaultExecutorType());
191     assertEquals(AutoMappingBehavior.FULL, configuration.getAutoMappingBehavior());
192     assertTrue(configuration.isCallSettersOnNulls());
193     assertEquals(defaultStatementTimeout, configuration.getDefaultStatementTimeout());
194     assertTrue(configuration.isMapUnderscoreToCamelCase());
195     assertEquals(defaultFetchSize, configuration.getDefaultFetchSize());
196   }

198   @Test
199   void get_ConfigurationSettingOverwritesNamed() throws Throwable {
200     injector = Guice.createInjector(new AbstractModule() {
201       @Override
202       protected void configure() {
203         bind(Environment.class).toInstance(environment);
204         bind(DataSource.class).toInstance(dataSource);
205         bind(Configuration.class).toProvider(configurationProvider);
206         bindConstant().annotatedWith(Names.named("mybatis.configuration.lazyLoadingEnabled")).to(true);
207       }
208     });
209     configurationProvider.addConfigurationSetting(new ConfigurationSetting() {
210       @Override
211       public void applyConfigurationSetting(Configuration configuration) {
212         configuration.setLazyLoadingEnabled(false);
213       }
214     });
215 
216     Configuration configuration = injector.getInstance(Configuration.class);
217 
218     configuration.getMappedStatementNames(); // Test that configuration is valid.
219     assertFalse(configuration.isLazyLoadingEnabled());
220   }
