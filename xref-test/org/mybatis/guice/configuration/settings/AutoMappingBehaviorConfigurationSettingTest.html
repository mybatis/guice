13  package org.mybatis.guice.configuration.settings;
14  
15  import static org.mockito.Mockito.verify;
16  
17  import org.apache.ibatis.session.AutoMappingBehavior;
18  import org.apache.ibatis.session.Configuration;
19  import org.junit.Rule;
20  import org.junit.Test;
21  import org.mockito.Mock;
22  import org.mockito.junit.MockitoJUnit;
23  import org.mockito.junit.MockitoRule;
24  
25  public class AutoMappingBehaviorConfigurationSettingTest {
26    @Mock
27    private Configuration configuration;
28    @Rule
29    public MockitoRule mockitoRule = MockitoJUnit.rule();
30  
31    @Test
32    public void applyConfigurationSetting_Full() {
33      AutoMappingBehaviorConfigurationSetting setting = new AutoMappingBehaviorConfigurationSetting(
34          AutoMappingBehavior.FULL);
35      setting.applyConfigurationSetting(configuration);
36      verify(configuration).setAutoMappingBehavior(AutoMappingBehavior.FULL);
37    }
38  
39    @Test
40    public void applyConfigurationSetting_None() {
41      AutoMappingBehaviorConfigurationSetting setting = new AutoMappingBehaviorConfigurationSetting(
42          AutoMappingBehavior.NONE);
43      setting.applyConfigurationSetting(configuration);
44      verify(configuration).setAutoMappingBehavior(AutoMappingBehavior.NONE);
45    }
46  
47    @Test
48    public void applyConfigurationSetting_Partial() {
49      AutoMappingBehaviorConfigurationSetting setting = new AutoMappingBehaviorConfigurationSetting(
50          AutoMappingBehavior.PARTIAL);
51      setting.applyConfigurationSetting(configuration);
52      verify(configuration).setAutoMappingBehavior(AutoMappingBehavior.PARTIAL);
53    }
54  }
