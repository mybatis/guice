16  package org.mybatis.guice.transactional;
17  
18  import static org.junit.jupiter.api.Assertions.assertEquals;
19  import static org.junit.jupiter.api.Assertions.assertThrows;
20  import static org.mockito.Mockito.mock;
21  import static org.mockito.Mockito.verify;
22  import static org.mockito.Mockito.when;
23  
24  import jakarta.transaction.Status;
25  import jakarta.transaction.Transaction;
26  import jakarta.transaction.TransactionManager;
27  
28  import java.lang.reflect.Field;
29  import java.lang.reflect.Method;
30  
31  import javax.transaction.xa.XAResource;
32  
33  import org.aopalliance.intercept.MethodInvocation;
34  import org.junit.jupiter.api.BeforeEach;
35  import org.junit.jupiter.api.Test;
36  
37  class TxTransactionalMethodInterceptorTest {
38  
39    private TxTransactionalMethodInterceptor interceptor;
40    private TransactionManager manager;
41    private Transaction transaction;
42    private XAResource xaResource;
43  
44    @BeforeEach
45    void beforeTest() throws Exception {
46      interceptor = new TxTransactionalMethodInterceptor();
47      manager = mock(TransactionManager.class);
48      transaction = mock(Transaction.class);
49      xaResource = mock(XAResource.class);
50  
51      setField(interceptor, "manager", manager);
52      setField(interceptor, "xaResourceProvider", (jakarta.inject.Provider<XAResource>) () -> xaResource);
53    }
54  
55    @Test
56    void invoke_WithRequiredTransaction_Commits() throws Throwable {
57      Method method = RequiredMethod.class.getMethod("tx");
58      MethodInvocation invocation = mock(MethodInvocation.class);
59      when(invocation.getMethod()).thenReturn(method);
60      when(invocation.proceed()).thenReturn("ok");
61      when(manager.getStatus()).thenReturn(Status.STATUS_NO_TRANSACTION, Status.STATUS_ACTIVE);
62      when(manager.getTransaction()).thenReturn(transaction);
63      when(transaction.enlistResource(xaResource)).thenReturn(true);
64  
65      Object result = interceptor.invoke(invocation);
66  
67      assertEquals("ok", result);
68      verify(manager).begin();
69      verify(transaction).enlistResource(xaResource);
70      verify(manager).commit();
71    }
72  
73    @Test
74    void invoke_WithRollbackOnlyAnnotation_RollsBack() throws Throwable {
75      Method method = RollbackOnlyMethod.class.getMethod("tx");
76      MethodInvocation invocation = mock(MethodInvocation.class);
77      when(invocation.getMethod()).thenReturn(method);
78      when(invocation.proceed()).thenReturn("ok");
79      when(manager.getStatus()).thenReturn(Status.STATUS_NO_TRANSACTION, Status.STATUS_MARKED_ROLLBACK);
80      when(manager.getTransaction()).thenReturn(transaction);
81      when(transaction.enlistResource(xaResource)).thenReturn(true);
82  
83      interceptor.invoke(invocation);
84  
85      verify(manager).setRollbackOnly();
86      verify(manager).rollback();
87    }
88  
89    @Test
90    void invoke_WithRuntimeException_MarksRollbackAndRethrows() throws Throwable {
91      Method method = RequiredMethod.class.getMethod("tx");
92      MethodInvocation invocation = mock(MethodInvocation.class);
93      RuntimeException failure = new RuntimeException("failure");
94      when(invocation.getMethod()).thenReturn(method);
95      when(invocation.proceed()).thenThrow(failure);
96      when(manager.getStatus()).thenReturn(Status.STATUS_NO_TRANSACTION, Status.STATUS_MARKED_ROLLBACK);
97      when(manager.getTransaction()).thenReturn(transaction);
98      when(transaction.enlistResource(xaResource)).thenReturn(true);
99  
100     RuntimeException thrown = assertThrows(RuntimeException.class, () -> interceptor.invoke(invocation));
101     assertEquals("failure", thrown.getMessage());
102     verify(manager).setRollbackOnly();
103     verify(manager).rollback();
104   }
105 
106   @Test
107   void invoke_WithoutManager_SkipsTxHandling() throws Throwable {
108     setField(interceptor, "manager", null);
109 
110     Method method = RequiredMethod.class.getMethod("tx");
111     MethodInvocation invocation = mock(MethodInvocation.class);
112     when(invocation.getMethod()).thenReturn(method);
113     when(invocation.proceed()).thenReturn("ok");
114 
115     assertEquals("ok", interceptor.invoke(invocation));
116   }
117 
118   @Test
119   void invoke_WithoutManager_UsesClassLevelAnnotation() throws Throwable {
120     setField(interceptor, "manager", null);
121 
122     Method method = ClassLevelRequiredMethod.class.getMethod("tx");
123     MethodInvocation invocation = mock(MethodInvocation.class);
124     when(invocation.getMethod()).thenReturn(method);
125     when(invocation.proceed()).thenReturn("ok");
126 
127     assertEquals("ok", interceptor.invoke(invocation));
128   }
129 
130   @Test
131   void invoke_WithoutManager_RethrowsInvocationError() throws Throwable {
132     setField(interceptor, "manager", null);
133 
134     Method method = RequiredMethod.class.getMethod("tx");
135     MethodInvocation invocation = mock(MethodInvocation.class);
136     when(invocation.getMethod()).thenReturn(method);
137     when(invocation.proceed()).thenThrow(new IllegalStateException("boom"));
138 
139     IllegalStateException thrown = assertThrows(IllegalStateException.class, () -> interceptor.invoke(invocation));
140     assertEquals("boom", thrown.getMessage());
141   }
142 
143   private static void setField(Object target, String fieldName, Object value) throws Exception {
144     Field field = target.getClass().getDeclaredField(fieldName);
145     field.setAccessible(true);
146     field.set(target, value);
147   }
148 
149   private static class RequiredMethod {
150     @Transactional(Transactional.TxType.REQUIRED)
151     public String tx() {
152       return "ok";
153     }
154   }
155 
156   private static class RollbackOnlyMethod {
157     @Transactional(value = Transactional.TxType.REQUIRED, rollbackOnly = true)
158     public String tx() {
159       return "ok";
160     }
161   }
162 
163   @Transactional(Transactional.TxType.REQUIRED)
164   private static class ClassLevelRequiredMethod {
165     public String tx() {
166       return "ok";
167     }
168   }
169 }
