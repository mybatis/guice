16  package org.mybatis.guice;
17  
18  import com.google.inject.Module;
19  
20  import java.sql.SQLException;
21  import java.util.ArrayList;
22  import java.util.List;
23  import java.util.Properties;
24  
25  import org.apache.ibatis.mapping.VendorDatabaseIdProvider;
26  import org.apache.ibatis.transaction.jdbc.JdbcTransactionFactory;
27  import org.mybatis.guice.datasource.builtin.PooledDataSourceProvider;
28  import org.mybatis.guice.datasource.helper.JdbcHelper;
29  
30  public final class GuiceTestExtension extends AbstractGuiceTestExtension {
31  
32    public GuiceTestExtension() throws SQLException {
33      super();
34    }
35  
36    @Override
37    protected List<Module> createMyBatisModule() {
38      List<Module> modules = new ArrayList<Module>(3);
39  
40      modules.add(JdbcHelper.HSQLDB_IN_MEMORY_NAMED);
41      modules.add(new MyBatisModule() {
42  
43        @Override
44        protected void initialize() {
45          bindDataSourceProviderType(PooledDataSourceProvider.class);
46          bindTransactionFactoryType(JdbcTransactionFactory.class);
47          addMapperClass(ContactMapper.class);
48          handleType(CustomType.class).with(CustomLongTypeHandler.class);
49          handleType(Address.class).with(AddressTypeHandler.class);
50          addInterceptorClass(CountUpdateInterceptor.class);
51          addTypeHandlerClass(ContactIdTypeHandler.class);
52          addTypeHandlerClass(ContactNameTypeHandler.class);
53          bindDatabaseIdProvider(VendorDatabaseIdProvider.class);
54          defaultStatementTimeout(null);
55        }
56  
57      });
58      modules.add(new MyBatisModule() {
59  
60        @Override
61        protected void initialize() {
62          addMapperClass(ContactMapper.class);
63          handleType(CustomType.class).with(CustomLongTypeHandler.class);
64          handleType(Address.class).with(AddressTypeHandler.class);
65          addTypeHandlerClass(ContactIdTypeHandler.class);
66          addTypeHandlerClass(ContactNameTypeHandler.class);
67        }
68  
69      });
70  
71      return modules;
72    }
73  
74    @Override
75    protected Properties createTestProperties() {
76      final Properties myBatisProperties = new Properties();
77      myBatisProperties.setProperty("mybatis.environment.id", "test");
78      myBatisProperties.setProperty("JDBC.username", "sa");
79      myBatisProperties.setProperty("JDBC.password", "");
80      myBatisProperties.setProperty("JDBC.autoCommit", "false");
81      return myBatisProperties;
82    }
83  }
