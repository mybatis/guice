16  package org.mybatis.guice;
17  
18  import jakarta.inject.Inject;
19  import jakarta.inject.Named;
20  
21  import java.io.StringReader;
22  import java.sql.Timestamp;
23  
24  import org.apache.ibatis.jdbc.ScriptRunner;
25  import org.apache.ibatis.session.SqlSession;
26  
27  public class CleanDatabaseRule {
28    private SqlSession sqlSession;
29    private Contact contact;
30    private Contact contactWithAddress;
31    private AddressConverter addressConverter;
32  
33    @Inject
34    private CleanDatabaseRule(SqlSession sqlSession, Contact contact,
35        @Named("contactWithAddress") Contact contactWithAddress, AddressConverter addressConverter) {
36      this.sqlSession = sqlSession;
37      this.contact = contact;
38      this.contactWithAddress = contactWithAddress;
39      this.addressConverter = addressConverter;
40    }
41  
42    public void evaluate() throws Exception {
43      ScriptRunner runner = new ScriptRunner(
44          sqlSession.getConfiguration().getEnvironment().getDataSource().getConnection());
45      try {
46        runner.setAutoCommit(true);
47        runner.setStopOnError(true);
48        runner.runScript(new StringReader("DELETE FROM contact; "
49            + "INSERT INTO contact (id, first_name, last_name, created) " + "VALUES (1, '" + contact.getFirstName()
50            + "', '" + contact.getLastName() + "', '" + new Timestamp(contactWithAddress.getCreated().getValue()) + "'); "
51            + "INSERT INTO contact (id, first_name, last_name, created, address) " + "VALUES (2, '"
52            + contactWithAddress.getFirstName() + "', '" + contactWithAddress.getLastName() + "', '"
53            + new Timestamp(contactWithAddress.getCreated().getValue()) + "', '"
54            + addressConverter.convert(contactWithAddress.getAddress()) + "'); "));
55        contact.setId(1);
56        contactWithAddress.setId(2);
57      } finally {
58        runner.closeConnection();
59      }
60    }
61  
62  }
