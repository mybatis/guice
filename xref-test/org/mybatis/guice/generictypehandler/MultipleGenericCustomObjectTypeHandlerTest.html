16  package org.mybatis.guice.generictypehandler;
17  
18  import static com.google.inject.name.Names.bindProperties;
19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertNotNull;
21  
22  import java.util.Properties;
23  
24  import org.apache.ibatis.session.SqlSessionFactory;
25  import org.apache.ibatis.transaction.jdbc.JdbcTransactionFactory;
26  import org.apache.ibatis.type.TypeHandler;
27  import org.junit.Test;
28  import org.mybatis.guice.MyBatisModule;
29  import org.mybatis.guice.datasource.builtin.PooledDataSourceProvider;
30  import org.mybatis.guice.datasource.helper.JdbcHelper;
31  
32  import com.google.inject.Guice;
33  import com.google.inject.Injector;
34  import com.google.inject.TypeLiteral;
35  
36  public class MultipleGenericCustomObjectTypeHandlerTest {
37      @Test
38      @SuppressWarnings({ "rawtypes" })
39      public void genericTypeHandler() {
40          Injector injector = Guice.createInjector(JdbcHelper.HSQLDB_IN_MEMORY_NAMED, new MyBatisModule() {
41              @Override
42              protected void initialize() {
43                  final Properties myBatisProperties = new Properties();
44                  myBatisProperties.setProperty("mybatis.environment.id", "test");
45                  myBatisProperties.setProperty("JDBC.username", "sa");
46                  myBatisProperties.setProperty("JDBC.password", "");
47                  myBatisProperties.setProperty("JDBC.autoCommit", "false");
48                  bindProperties(binder(), myBatisProperties);
49                  bindDataSourceProviderType(PooledDataSourceProvider.class);
50                  bindTransactionFactoryType(JdbcTransactionFactory.class);
51                  handleType(CustomObject.class).withProvidedTypeHandler(new TypeLiteral<GenericCustomObjectTypeHandler<CustomObject>>(){});
52                  handleType(SubCustomObject.class).withProvidedTypeHandler(new TypeLiteral<GenericCustomObjectTypeHandler<SubCustomObject>>(){});
53              }
54          });
55          SqlSessionFactory factory = injector.getInstance(SqlSessionFactory.class);
56          TypeHandler<?> typeHandler = factory.getConfiguration().getTypeHandlerRegistry().getTypeHandler(CustomObject.class);
57          assertEquals(GenericCustomObjectTypeHandler.class, typeHandler.getClass());
58          GenericCustomObjectTypeHandler genericEnumTypeHandler = (GenericCustomObjectTypeHandler)typeHandler;
59          assertEquals(CustomObject.class, genericEnumTypeHandler.getType());
60          assertNotNull(genericEnumTypeHandler.getInjectedObject());
61          typeHandler = factory.getConfiguration().getTypeHandlerRegistry().getTypeHandler(SubCustomObject.class);
62          assertEquals(GenericCustomObjectTypeHandler.class, typeHandler.getClass());
63          genericEnumTypeHandler = (GenericCustomObjectTypeHandler)typeHandler;
64          assertEquals(SubCustomObject.class, genericEnumTypeHandler.getType());
65          assertNotNull(genericEnumTypeHandler.getInjectedObject());
66      }
67  }
