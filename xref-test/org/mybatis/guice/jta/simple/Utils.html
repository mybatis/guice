18  import com.arjuna.ats.arjuna.recovery.RecoveryManager;
19  import com.arjuna.ats.internal.jta.transaction.arjunacore.TransactionSynchronizationRegistryImple;
20  
21  import io.agroal.api.AgroalDataSource;
22  import io.agroal.api.configuration.supplier.AgroalDataSourceConfigurationSupplier;
23  import io.agroal.api.security.NamePrincipal;
24  import io.agroal.api.security.SimplePassword;
25  import io.agroal.narayana.NarayanaTransactionIntegration;
26  
27  import jakarta.transaction.TransactionManager;
28  import jakarta.transaction.TransactionSynchronizationRegistry;
29  
30  import java.util.Properties;
31  
32  import javax.naming.Context;
33  import javax.naming.InitialContext;
34  import javax.sql.DataSource;
35  
36  import org.mybatis.guice.multidstest.MockInitialContextFactory;
37  
38  public class Utils {
39  
40    /**
41     * This method sets up a mock JNDI context with a transaction manager and two XA datasources
42     *
43     * @throws Exception
44     */
45    public static void setupMockJNDI() throws Exception {
46      Properties properties = new Properties();
47      properties.setProperty(Context.INITIAL_CONTEXT_FACTORY, MockInitialContextFactory.class.getName());
48      InitialContext ic = new InitialContext(properties);
49  
50      TransactionManager tm = com.arjuna.ats.jta.TransactionManager.transactionManager();
51      TransactionSynchronizationRegistry transactionSynchronizationRegistry = new TransactionSynchronizationRegistryImple();
52      // Initialization of recovery manager
53      RecoveryManager recoveryManager = RecoveryManager.manager();
54      recoveryManager.initialize();
55  
56      ic.bind("jakarta.transaction.TransactionManager", tm);
57  
58      DataSource ads1 = AgroalDataSource
59          .from(new AgroalDataSourceConfigurationSupplier().connectionPoolConfiguration(cp -> cp.maxSize(10)
60              .connectionFactoryConfiguration(cf -> cf.jdbcUrl("jdbc:hsqldb:mem:schema1")
61                  .principal(new NamePrincipal("sa")).credential(new SimplePassword("")))
62              .transactionIntegration(
63                  new NarayanaTransactionIntegration(tm, transactionSynchronizationRegistry, "java:/agroalds1", false))));
64  
65      ic.bind("java:comp/env/jdbc/DS1", ads1);
66  
67      DataSource ads2 = AgroalDataSource
68          .from(new AgroalDataSourceConfigurationSupplier().connectionPoolConfiguration(cp -> cp.maxSize(10)
69              .connectionFactoryConfiguration(cf -> cf.jdbcUrl("jdbc:hsqldb:mem:schema2")
70                  .principal(new NamePrincipal("sa")).credential(new SimplePassword("")))
71              .transactionIntegration(
72                  new NarayanaTransactionIntegration(tm, transactionSynchronizationRegistry, "java:/agroalds2", false))));
73  
74      ic.bind("java:comp/env/jdbc/DS2", ads2);
75    }
76  }
