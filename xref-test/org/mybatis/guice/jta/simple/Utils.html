16  package org.mybatis.guice.jta.simple;
17  
18  import com.arjuna.ats.arjuna.recovery.RecoveryManager;
19  
20  import io.agroal.api.AgroalDataSource;
21  import io.agroal.api.configuration.supplier.AgroalDataSourceConfigurationSupplier;
22  import io.agroal.api.security.NamePrincipal;
23  import io.agroal.api.security.SimplePassword;
24  import io.agroal.narayana.NarayanaTransactionIntegration;
25  
26  import jakarta.transaction.TransactionManager;
27  import jakarta.transaction.TransactionSynchronizationRegistry;
28  
29  import java.util.Properties;
30  
31  import javax.naming.Context;
32  import javax.naming.InitialContext;
33  import javax.sql.DataSource;
34  
35  import org.mybatis.guice.multidstest.MockInitialContextFactory;
36  
37  public class Utils {
38  
39    /**
40     * This method sets up a mock JNDI context with a transaction manager and two XA datasources
41     *
42     * @throws Exception
43     */
44    public static void setupMockJNDI() throws Exception {
45      Properties properties = new Properties();
46      properties.setProperty(Context.INITIAL_CONTEXT_FACTORY, MockInitialContextFactory.class.getName());
47      InitialContext ic = new InitialContext(properties);
48  
49      TransactionManager tm = com.arjuna.ats.jta.TransactionManager.transactionManager();
50      TransactionSynchronizationRegistry transactionSynchronizationRegistry = new com.arjuna.ats.internal.jta.transaction.arjunacore.TransactionSynchronizationRegistryImple();
51      // Initialization of recovery manager
52      RecoveryManager recoveryManager = com.arjuna.ats.arjuna.recovery.RecoveryManager.manager();
53      recoveryManager.initialize();
54  
55      ic.bind("jakarta.transaction.TransactionManager", tm);
56  
57      DataSource ads1 = AgroalDataSource
58          .from(new AgroalDataSourceConfigurationSupplier().connectionPoolConfiguration(cp -> cp.maxSize(10)
59              .connectionFactoryConfiguration(cf -> cf.jdbcUrl("jdbc:hsqldb:mem:schema1")
60                  .principal(new NamePrincipal("sa")).credential(new SimplePassword("")))
61              .transactionIntegration(
62                  new NarayanaTransactionIntegration(tm, transactionSynchronizationRegistry, "java:/agroalds1", false))));
63  
64      ic.bind("java:comp/env/jdbc/DS1", ads1);
65  
66      DataSource ads2 = AgroalDataSource
67          .from(new AgroalDataSourceConfigurationSupplier().connectionPoolConfiguration(cp -> cp.maxSize(10)
68              .connectionFactoryConfiguration(cf -> cf.jdbcUrl("jdbc:hsqldb:mem:schema2")
69                  .principal(new NamePrincipal("sa")).credential(new SimplePassword("")))
70              .transactionIntegration(
71                  new NarayanaTransactionIntegration(tm, transactionSynchronizationRegistry, "java:/agroalds2", false))));
72  
73      ic.bind("java:comp/env/jdbc/DS2", ads2);
74    }
75  }
